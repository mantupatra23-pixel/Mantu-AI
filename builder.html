/* ADD FILE */
function addFile(){
  let name=document.getElementById("fileName").value
  if(name){
    files[name]=""
    openFile(name)
  }
}

/* SAVE */
async function save(){
  let name=document.getElementById("projectName").value

  await fetch("/save_project",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({name,files})
  })

  alert("Saved")
}

/* LOAD */
async function load(){
  let name=document.getElementById("projectName").value

  let res=await fetch("/load_project/"+name)
  let data=await res.json()

  files=data.files
  openFile(Object.keys(files)[0])
}

/* ENV SAVE */
async function saveEnv(){
  let project=document.getElementById("projectName").value
  let raw=document.getElementById("envInput").value

  let env={}
  raw.split("\n").forEach(line=>{
    let [k,v]=line.split("=")
    if(k && v) env[k.trim()]=v.trim()
  })

  await fetch("/save_env",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({project,env})
  })

  alert("ENV Saved")
}

/* CHAT */
async function chat(){
  let msg=document.getElementById("chatInput").value

  let res=await fetch("/chat",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({message:msg})
  })

  let data=await res.json()

  files["index.html"] += "\n"+data.reply
  openFile("index.html")
}

/* DOWNLOAD */
async function download(){
  let name="app"

  let res=await fetch("/export_zip",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({name,files})
  })

  let blob=await res.blob()
  let url=URL.createObjectURL(blob)

  let a=document.createElement("a")
  a.href=url
  a.download="app.zip"
  a.click()
}

/* DEPLOY */
async function deploy(){
  await fetch("/deploy/github",{
    method:"POST",
    headers:{"Content-Type":"application/json"},
    body:JSON.stringify({name:"app",files})
  })

  alert("Deploy 🚀")
}
